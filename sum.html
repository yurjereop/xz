Нахождение суммы элементов в массиве

Этот алгоритм заключается в создании переменной для суммы и в последующем добавлении к ней каждого элемента массива 

program sum;
var
    arr: array[1..100] of integer;
    n, sum, i: integer; 
begin
    writeln('Количество элементов: ');
    readln(n); 
    writeln('Укажите значения: ');
    for i := 1 to n do
        readln(arr[i]);
    sum := 0;
    for i := 1 to n do
        sum := sum + arr[i];
     writeln('Сумма: ', sum);
 end.

Подсчёт ненулевых элементов в массиве

Алгоритм состоит в сравнении каждого элемента массива с нулём, и при каждом выполенении условия увеличивается переменная-счётчик на 1. 

program null;
var
    arr: array[1..100] of integer;
    n, i, k: integer; 
begin
    writeln('Количество элементов: ');
    readln(n); 
    writeln('Укажите значения: ');
    for i := 1 to n do
        readln(arr[i]);
    k := 0;
    for i := 1 to n do
        if arr[i] <> 0 then
            k := k + 1;
    writeln('Количество ненулевых: ', k)
 end.

Удаление элемента из массива

Чтобы удалить элемент из массива, нужно все элементы справа от него сдивнуть влево до его позиции и уменьшить размер массива на один. 

program add;
var
    arr: array[1..100] of integer;
    n, pos, i: integer; 
begin
    writeln('Количество элементов: ');
    readln(n); 
    writeln('Укажите значения: ');
    for i := 1 to n do
      readln(arr[i]);
    writeln('Позиция удаления: ');
    readln(pos);
    for i:= pos to n do 
      arr[i] := arr[i+1];
    n := n - 1;
    for i:= 1 to n do 
      write(arr[i], ' ')
end.
